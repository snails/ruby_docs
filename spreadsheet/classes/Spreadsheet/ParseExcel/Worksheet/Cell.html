def to_s(target_encoding=nil)
  if(target_encoding)
    begin
      Iconv.new(target_encoding, @encoding).iconv(@value)
    rescue
      Iconv.new(target_encoding, 'ascii').iconv(@value.to_s)
    end
  else
    @value.to_s
  end
end

def type
  if @format && (@format.date? || @format.time?)
    :date
  elsif @value.is_a?(Numeric)
    :numeric
  else
    :text
  end
end
